import PropTable from '../../../components/PropTable.astro';

Nodes are `{ id, label?, color? }`; links are `{ source, target, value }`. `computeSankey` (built on the DAG layering) sizes nodes by throughput and ribbons by value.

## Basic

```tsx
import { SankeyChart } from 'goldenchart';

<SankeyChart
  width={520}
  height={320}
  nodes={[
    { id: 'visits', label: 'Visits' },
    { id: 'signups', label: 'Signups' },
    { id: 'paid', label: 'Paid' },
    { id: 'churned', label: 'Churned' },
  ]}
  links={[
    { source: 'visits', target: 'signups', value: 40 },
    { source: 'signups', target: 'paid', value: 24 },
    { source: 'signups', target: 'churned', value: 16 },
  ]}
/>;
```
