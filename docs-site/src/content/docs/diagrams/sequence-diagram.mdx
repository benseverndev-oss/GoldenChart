import PropTable from '../../../components/PropTable.astro';

`actors` run across the top with lifelines dropping down; `messages` are ordered horizontal arrows between them. Reply messages dash; self-messages loop back. Message labels get a solid knockout so they stay legible over the lifelines.

```tsx
import { SequenceDiagram } from 'goldenchart';

<SequenceDiagram
  width={520}
  height={320}
  actors={[
    { id: 'user', label: 'User' },
    { id: 'app', label: 'App' },
    { id: 'db', label: 'DB' },
  ]}
  messages={[
    { from: 'user', to: 'app', label: 'click' },
    { from: 'app', to: 'db', label: 'query' },
    { from: 'db', to: 'app', label: 'rows', kind: 'reply' },
    { from: 'app', to: 'user', label: 'render', kind: 'reply' },
  ]}
/>;
```
