import PropTable from '../../../components/PropTable.astro';

A thin wrapper over [Diagram](/diagrams/diagram/) preset to a radial layout: the first node sits at the centre and children fan out on straight spokes.

```tsx
import { MindMap } from 'goldenchart';

<MindMap
  width={480}
  height={400}
  nodes={[
    { id: 'root', label: 'Release' },
    { id: 'docs', label: 'Docs' },
    { id: 'tests', label: 'Tests' },
    { id: 'ci', label: 'CI' },
  ]}
  edges={[
    { from: 'root', to: 'docs' },
    { from: 'root', to: 'tests' },
    { from: 'root', to: 'ci' },
  ]}
/>;
```
